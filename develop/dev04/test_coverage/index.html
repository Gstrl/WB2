// –∫–∞–∂–¥—ã–π —ç–ª–µ–º–µ–Ω—Ç –∫–æ—Ç–æ—Ä–æ–≥–æ - —Å–ª–æ–≤–æ –Ω–∞ —Ä—É—Å—Å–∫–æ–º —è–∑—ã–∫–µ –≤ –∫–æ–¥–∏—Ä–æ–≤–∫–µ utf8.
// –≤ –Ω–µ–∫–æ—Ç–æ—Ä—ã–π –º–µ—Å—Ç–∞—Ö –¥–ª—è –ø–æ–ª—É—á–µ–Ω–∏—è –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ —Å–∏–º–≤–æ–ª–æ–≤, —è –¥–µ–ª—é
// –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –±–∞–π—Ç –Ω–∞ 2, –æ—Ç—Å—é–¥–∞ –±–µ—Ä–µ—Ç—Å—è Magic number

// Set  setSymbols - —Å–∏–º–≤–æ–ª—ã –º–Ω–æ–∂–µ—Å—Ç–≤–∞ –∏ –∏—Ö –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ, setElem - —ç–ª–µ–º–µ–Ω—Ç—ã –º–Ω–æ–∂–µ—Å—Ç–≤–∞
type Set struct {
        setSymbols map[rune]int
        setElem    []string
}

// CheckSet –ø—Ä–æ–≤–µ—Ä—è–µ—Ç –≤—Ö–æ–¥–∏—Ç –ª–∏ —Å–ª–æ–≤–æ –≤ –º–Ω–æ–∂–µ—Å—Ç–≤–æ
func (s *Set) CheckSet(str string) bool {
        setStr := make(map[rune]int, len(str)/2)
        for _, v := range str {
                _, ok := setStr[v]
                if ok {
                        setStr[v]++
                } else {
                        setStr[v] = 1
                }
        }

        // –µ—Å–ª–∏ —Å–∏–º–≤–æ–ª—ã –∏ –∏—Ö –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –≤ —Å–ª–æ–≤–µ —Å–æ–≤–ø–∞–¥–∞–µ—Ç —Å —Å–∏–º–≤–æ–ª–∞–º–∏ –º–Ω–æ–∂–µ—Å—Ç–≤–∞ –≤–æ–∑–≤—Ä–∞—â–∞–µ–º true
        for _, v := range str {
                _, ok := s.setSymbols[v]
                if ok {
                        if setStr[v] != s.setSymbols[v] {
                                return false
                        }
                } else {
                        return false
                }
        }
        return true
}

// AppendSet –¥–æ–±–∞–≤–ª—è–µ—Ç —Å–ª–æ–≤–æ –∫ —ç–ª–µ–º–µ–Ω—Ç–∞–º –º–Ω–æ–∂–µ—Å—Ç–≤–∞
func (s *Set) AppendSet(str string) {
        s.setElem = append(s.setElem, str)
}

// newSet —Å–æ–∑–¥–∞–µ—Ç –Ω–æ–≤–æ–µ –º–Ω–æ–∂–µ—Å—Ç–≤–æ (—Å–∏–º–≤–æ–ª—ã –º–Ω–æ–∂–µ—Å—Ç–≤–∞ –∏ –ø–µ—Ä–≤—ã–π —ç–ª–µ–º–µ–Ω—Ç)
func newSet(str string) Set {
        setElem := make([]string, 1, 2)
        setElem[0] = str
        setSymbols := make(map[rune]int, len(str)/2)
        for _, v := range str {
                _, ok := setSymbols[v]
                if ok {
                        setSymbols[v]++
                } else {
                        setSymbols[v] = 1
                }
        }
        return Set{
                setElem:    setElem,
                setSymbols: setSymbols,
        }
}

// MakeSet –≥—Ä—É–ø–ø–∏—Ä—É–µ—Ç —Å–ª–∞–π—Å —Å—Ç—Ä–æ–∫ –≤ –º–Ω–æ–∂–µ—Å—Ç–≤–∞
func MakeSet(arrStr []string) map[string][]string {
        if len(arrStr) == 0 {
                return nil
        }
        // –ø—Ä–∏–≤–µ–¥–µ–Ω–∏–µ –∫ –Ω–∏–∂–Ω–µ–º—É —Ä–µ–≥–∏—Å—Ç—Ä—É
        for i := 0; i <= len(arrStr)-1; i++ {
                arrStr[i] = strings.ToLower(arrStr[i])
        }

        // —Å–æ–∑–¥–∞–µ–º —Å–ª–∞–π—Å –º–Ω–æ–∂–µ—Å—Ç–≤ –∏ –¥–æ–±–∞–≤–ª—è–µ–º –ø–µ—Ä–≤–æ–µ –º–Ω–æ–∂–µ—Å—Ç–≤–æ
        arrSet := make([]Set, 0)
        arrSet = append(arrSet, newSet(arrStr[0]))

        // –µ—Å–ª–∏ –Ω–µ –ø–æ–ø–∞–ª–∏ –Ω–∏ –≤ –æ–¥–Ω–æ –º–Ω–æ–∂–µ—Å—Ç–≤–æ —Å–æ–∑–¥–∞–µ–º –Ω–æ–≤–æ–µ
        for i := 1; i <= len(arrStr)-1; i++ {
                // –µ—Å–ª–∏ –≤ —Å–ª–æ–≤–µ <=1 –æ–Ω–æ –Ω–µ –º–æ–∂–µ—Ç –±—ã—Ç—å –º–Ω–æ–∂–µ—Å—Ç–≤–æ–º
                if len(arrStr[i]) <= 1 {
                        continue
                }

                hitSet := false

                for ii := 0; ii <= len(arrSet)-1; ii++ {
                        if arrSet[ii].CheckSet(arrStr[i]) {
                                hitSet = true
                                arrSet[ii].AppendSet(arrStr[i])
                                break
                        }
                }
                if !hitSet {
                        arrSet = append(arrSet, newSet(arrStr[i]))
                }
        }
        return resMap(arrSet)
}

// resMap –ø—Ä–æ–≤–µ—Ä—è–µ—Ç –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –º–Ω–æ–∂–µ—Å—Ç–≤–∞ –∏ –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç —Ä–µ–∑—É–ª—å—Ç–∞—Ç –≥—Ä—É–ø–ø–∏—Ä–æ–≤–∫–∏
func resMap(s []Set) map[string][]string {
        res := make(map[string][]string)
        for i := 0; i <= len(s)-1; i++ {
                if len(s[i].setElem) == 1 {
                        continue
                }
                res[s[i].setElem[0]] = s[i].setElem
        }
        return res
}
